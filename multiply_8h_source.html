55  template<typename Upper, typename Lower>
56  struct long_multiply<duplex_integer<Upper, Lower>> {
57  using _duplex_integer = duplex_integer<Upper, Lower>;

64  duplex_integer<RhsUpper, RhsLower> const& rhs) const -> result_type

85  template<typename LhsUpper, typename LhsLower, typename RhsUpper, typename RhsLower>
86  [[nodiscard]] static constexpr auto multiply_components(
87  LhsUpper const& lhs_upper, LhsLower const& lhs_lower, RhsUpper const& rhs_upper,
88  RhsLower const& rhs_lower) -> result_type
89  {
90  auto const upper_upper{_impl::long_multiply<Upper>{}(lhs_upper, rhs_upper)};
91  auto const upper_lower{_impl::long_multiply<Upper>{}(lhs_upper, rhs_lower)};
92  auto const lower_upper{_impl::long_multiply<Upper>{}(lhs_lower, rhs_upper)};
93  auto const lower_lower{_impl::long_multiply<Lower>{}(lhs_lower, rhs_lower)};
94  auto const upper{_impl::sensible_left_shift<result_type>(
95  upper_upper,
96  digits<LhsLower> + digits<RhsLower>)};
97  auto const mid{
98  (result_type{upper_lower} << digits<LhsLower>)+(result_type{lower_upper} << digits<RhsLower>)};
99  auto const lower{lower_lower};
100  return upper + mid + lower;
101  }
102  };
103  }

114  _duplex_integer const& lhs, _duplex_integer const& rhs) const -> _duplex_integer

117  }
118 
119  [[nodiscard]] static constexpr auto multiply_components(
120  Upper const& lhs_upper, Lower const& lhs_lower, Upper const& rhs_upper,
121  Lower const& rhs_lower) -> _duplex_integer
122  {
123  auto const upper_upper{_impl::long_multiply<Upper>{}(lhs_upper, rhs_upper)};
124  auto const upper_lower{_impl::long_multiply<Upper>{}(lhs_upper, rhs_lower)};
125  auto const lower_upper{_impl::long_multiply<Upper>{}(lhs_lower, rhs_upper)};
126  auto const lower_lower{_impl::long_multiply<Lower>{}(lhs_lower, rhs_lower)};
127  auto const upper{_impl::sensible_left_shift<_duplex_integer>(upper_upper, digits<Lower> * 2)};
128  auto const mid{(upper_lower + lower_upper) << digits<Lower>};
129  auto const lower{lower_lower};
130  return upper + mid + lower;
131  }
132  };
