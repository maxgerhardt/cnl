40  // duplex_integer<> / duplex_integer<>
41  template<typename Upper, typename Lower>
42  struct custom_operator<
43  _impl::divide_op,
44  op_value<_impl::duplex_integer<Upper, Lower>>,
45  op_value<_impl::duplex_integer<Upper, Lower>>> {
46  using _duplex_integer = _impl::duplex_integer<Upper, Lower>;
47  using _unsigned_duplex_integer = numbers::set_signedness_t<_duplex_integer, false>;
48 
49  [[nodiscard]] constexpr auto operator()(
50  _duplex_integer const& lhs, _duplex_integer const& rhs) const -> _duplex_integer
51  {
52  return (lhs < _duplex_integer{0}) ? (rhs < _duplex_integer{0})
53  ? non_negative_division(-lhs, -rhs)
54  : -non_negative_division(-lhs, rhs)
55  : (rhs < _duplex_integer{0}) ? -non_negative_division(lhs, -rhs)
56  : non_negative_division(lhs, rhs);
57  }

59  // lifted from:
60  // https://github.com/torvalds/linux/blob/5ac94332248ee017964ba368cdda4ce647e3aba7/lib/math/div64.c#L142
61  static constexpr auto non_negative_division(
62  _unsigned_duplex_integer const& dividend, _unsigned_duplex_integer const& divisor)
63  -> _unsigned_duplex_integer
64  {
65  auto const high = divisor.upper();
66  if (!high) {
67  return div_by_lower(dividend, divisor.lower());
68  }
69 
70  int n = fls(high);
71  auto quot = div_by_lower(dividend >> n, (divisor >> n).lower());
72 
73  if (quot) {
74  --quot;
75  }
76  if ((dividend - quot * divisor) >= divisor) {
77  ++quot;
78  }
79 
80  return quot;
81  }
82 
83  static constexpr auto fls(Upper n) -> int
84  {
85  auto half_digits = numeric_limits<_duplex_integer>::digits / 2;
86 
87  if (!n) {
88  return 0;
89  }
90  auto const msd = Upper{1} << (half_digits - 1);
91  for (int r = half_digits;; n <<= 1, r--) {
92  if (n & msd) {
93  return r;
94  }
95  }
96  };

98  // from Linux div64_32
99  static constexpr auto div_by_lower(
100  _unsigned_duplex_integer const& dividend, Lower const& divisor)
101  -> _unsigned_duplex_integer
102  {
103  _unsigned_duplex_integer rem = dividend;
104  _unsigned_duplex_integer b = divisor;
105  _unsigned_duplex_integer d = 1;
106 
107  using unsigned_upper = numbers::set_signedness_t<Upper, false>;
108  auto high = rem.upper();
109 
110  _unsigned_duplex_integer quot = 0;
111  if (static_cast<unsigned_upper>(high) >= divisor) {
112  high /= divisor; // NOLINT(clang-analyzer-core.DivideZero)
113  quot = _unsigned_duplex_integer{high, 0};
114  rem -= _unsigned_duplex_integer(high * divisor, 0);
115  }
116 
117  while (b < rem) {
118  b <<= 1;
119  d <<= 1;
120  }
121 
122  do {
123  if (rem >= b) {
124  rem -= b;
125  quot += d;
126  }
127  b >>= 1;
128  d >>= 1;
129  } while (d);
130 
131  return quot;
132  };
133  };
