44  struct constant {

55  template<typename S>
56  constexpr explicit constant(S const& init)
57  {
58  static_assert(
59  value <= cnl::numeric_limits<S>::max()

61  "initial value couldn't possibly represent value");
62  CNL_ASSERT(value == init);
63  }
