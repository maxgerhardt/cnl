61  template<typename LhsParam, typename RhsParam>
62  [[nodiscard]] constexpr auto step2(LhsParam const& lhs, RhsParam const& rhs) const
63  -> result_type
64  {
65  return (lhs < 0) ? -((_impl::abs(lhs) + (rhs - (lhs < 0 ? 1 : 0)) / 2) / rhs)
66  : +((_impl::abs(lhs) + (rhs - (lhs < 0 ? 1 : 0)) / 2) / rhs);
67  }
68  [[nodiscard]] constexpr auto step1(Lhs const& lhs, Rhs const& rhs) const -> result_type
69  {
70  return (rhs < 0) ? step2(-lhs, -rhs) : step2(lhs, rhs);
71  }
